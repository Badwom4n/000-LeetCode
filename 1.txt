class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {


        vector<int> TwoSum(2,-1);
        map<int,int> HashMap;
        

        for(int i=0;i<(nums.size());i++){


            auto c=HashMap.find(target-nums[i]);
            if(c!=HashMap.end()){
                TwoSum[1]=i;
                TwoSum[0]=c->second;
                break;
            }

            HashMap[nums[i]]=i;
        }
        return TwoSum;
    }
};
