class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {

        string output = "";
        char letter;

        for (int i = 0; i < strs[0].length(); i++) {
            letter = strs[0][i];
            for (int j = 1; j < strs.size(); j++) {

                if (i == strs[j].length() )
                    return output;

                if (strs[j][i] != letter)
                    return output;

            }
            output += letter;
        }

        return output;
    }
};
