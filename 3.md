```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {

        unordered_set<char> Set;
        int Num=0;
        int MaxNum;
        int left=0;

        for(int i=0;i<s.size();i++){

            while(Set.find(s[i])!=Set.end()){
                Set.erase(s[left]);
                Num--;
                left++;
            }

            Set.insert(s[i]);
            Num++;
            MaxNum=max(MaxNum,Num);
        }

        return MaxNum;
    }
};
```
